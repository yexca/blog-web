    List<? extends Terms.Bucket> buckets = brand.getBuckets();
    // ã‚¤ãƒ†ãƒ¬ãƒ¼ãƒˆ
    List<String> brandList = new ArrayList<>();
    for (Terms.Bucket bucket : buckets) {
        // ã‚­ãƒ¼ã‚’å–å¾—
        String key = bucket.getKeyAsString();
        brandList.add(key);
    }
    return brandList;
}
