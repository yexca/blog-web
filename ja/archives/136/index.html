class Context{ // è‡ªè²©æ©Ÿ
    private int count;
    private State state;

    public Context(){
        count = 3;
        state = new StateA();
    }

    public int getCount() {
        return count;

    public State getState() {
        return state;
    }

interface State{
    public void Handle(Context context);
